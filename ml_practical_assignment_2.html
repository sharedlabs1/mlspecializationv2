def generate_banking_dataset(n_samples=5000):
    """Generate realistic banking dataset for loan default prediction"""
    
    # Basic demographics
    data = {
        'customer_id': [f'CUST_{i:06d}' for i in range(n_samples)],
        'age': np.random.normal(45, 15, n_samples).clip(18, 80).astype(int),
        'annual_income': np.random.lognormal(10.5, 0.5, n_samples).clip(20000, 500000),
        'employment_length': np.random.exponential(5, n_samples).clip(0, 40),
        'education_level': np.random.choice(['High School', 'Bachelor', 'Master', 'PhD', 'Some College'], 
                                          n_samples, p=[0.3, 0.35, 0.2, 0.05, 0.1]),
        'marital_status': np.random.choice(['Single', 'Married', 'Divorced', 'Widowed'], 
                                         n_samples, p=[0.35, 0.45, 0.15, 0.05]),
        'dependents': np.random.poisson(1.2, n_samples).clip(0, 5)
    }
    
    # Financial information
    data['credit_score'] = (np.random.beta(2, 2, n_samples) * 350 + 300).astype(int)
    data['existing_loans'] = np.random.poisson(1.5, n_samples).clip(0, 5)
    data['monthly_debt'] = data['annual_income'] * np.random.uniform(0.1, 0.4, n_samples) / 12
    data['account_balance'] = np.random.lognormal(8, 1.5, n_samples).clip(100, 100000)
    
    # Digital engagement
    data['app_usage_days'] = np.random.poisson(15, n_samples).clip(0, 30)
    data['digital_transactions'] = np.random.poisson(25, n_samples).clip(0, 100)
    data['customer_tenure'] = np.random.exponential(3, n_samples).clip(0.1, 15)
    
    # Create target variable with realistic correlations
    risk_score = (
        -0.3 * (data['credit_score'] - 300) / 350 +
        0.2 * (data['monthly_debt'] / (data['annual_income'] / 12)) +
        -0.1 * data['customer_tenure'] +
        0.1 * data['existing_loans'] +
        np.random.normal(0, 0.3, n_samples)
    )
    
    data['default_risk'] = (risk_score > np.percentile(risk_score, 80)).astype(int)
    
    # Convert to DataFrame
    df = pd.DataFrame(data)
    
    # Introduce missing values realistically
    missing_patterns = {
        'employment_length': 0.05,
        'education_level': 0.03,
        'account_balance': 0.02,
        'app_usage_days': 0.08
    }
    
    for col, missing_rate in missing_patterns.items():
        missing_mask = np.random.random(n_samples) < missing_rate
        df.loc[missing_mask, col] = np.nan
    
    # Add some outliers
    outlier_indices = np.random.choice(n_samples, int(n_samples * 0.02), replace=False)
    df.loc[outlier_indices, 'annual_income'] *= np.random.uniform(5, 10, len(outlier_indices))
    
    return df

from sklearn.preprocessing import StandardScaler, MinMaxScaler, RobustScaler
from sklearn.preprocessing import LabelEncoder, OneHotEncoder
from sklearn.impute import SimpleImputer, KNNImputer
from sklearn.compose import ColumnTransformer
from sklearn.pipeline import Pipeline

class BankingDataPreprocessor:
    def __init__(self):
        self.preprocessing_pipeline = None
        self.feature_names = None
    
    def create_preprocessing_pipeline(self, df):
        """Create comprehensive preprocessing pipeline"""
        
        # Identify column types
        numeric_features = df.select_dtypes(include=[np.number]).columns.tolist()
        numeric_features = [col for col in numeric_features if col not in ['customer_id', 'default_risk']]
        
        categorical_features = df.select_dtypes(include=['object']).columns.tolist()
        categorical_features = [col for col in categorical_features if col != 'customer_id']
        
        print(f"Numeric features: {numeric_features}")
        print(f"Categorical features: {categorical_features}")
        
        # Numeric preprocessing pipeline
        numeric_pipeline = Pipeline([
            ('imputer', KNNImputer(n_neighbors=5)),
            ('scaler', RobustScaler())  # Robust to outliers
        ])
        
        # Categorical preprocessing pipeline
        categorical_pipeline = Pipeline([
            ('imputer', SimpleImputer(strategy='most_frequent')),
            ('encoder', OneHotEncoder(drop='first', sparse_output=False, handle_unknown='ignore'))
        ])
        
        # Combine preprocessing steps
        self.preprocessing_pipeline = ColumnTransformer([
            ('numeric', numeric_pipeline, numeric_features),
            ('categorical', categorical_pipeline, categorical_features)
        ])
        
        return self.preprocessing_pipeline
    
    def fit_transform(self, X, y=None):
        """Fit and transform the data"""
        return self.preprocessing_pipeline.fit_transform(X)
    
    def transform(self, X):
        """Transform new data"""
        return self.preprocessing_pipeline.transform(X)
    
    def get_feature_names(self):
        """Get feature names after preprocessing"""
        if self.preprocessing_pipeline is None:
            return None
        
        # Get numeric feature names
        numeric_features = self.preprocessing_pipeline.named_transformers_['numeric']
        numeric_feature_names = self.preprocessing_pipeline.transformers_[0][2]
        
        # Get categorical feature names
        categorical_transformer = self.preprocessing_pipeline.named_transformers_['categorical']
        categorical_feature_names = categorical_transformer.named_steps['encoder'].get_feature_names_out(
            self.preprocessing_pipeline.transformers_[1][2]
        )
        
        return list(numeric_feature_names) + list(categorical_feature_names)

# Feature Engineering
def create_engineered_features(df):
    """Create domain-specific engineered features"""
    df_eng = df.copy()
    
    # Financial ratios
    df_eng['debt_to_income_ratio'] = (df_eng['monthly_debt'] * 12) / df_eng['annual_income']
    df_eng['balance_to_income_ratio'] = df_eng['account_balance'] / df_eng['annual_income']
    df_eng['loans_per_tenure'] = df_eng['existing_loans'] / (df_eng['customer_tenure'] + 1)
    
    # Age-related features
    df_eng['age_group'] = pd.cut(df_eng['age'], bins=[0, 30, 45, 60, 100], 
                                labels=['Young', 'Middle', 'Senior', 'Elder'])
    
    # Digital engagement features
    df_eng['digital_engagement_score'] = (
        df_eng['app_usage_days'] * 0.3 + 
        df_eng['digital_transactions'] * 0.7
    )
    
    # Credit utilization proxy
    df_eng['credit_utilization'] = df_eng['monthly_debt'] / (df_eng['credit_score'] / 100)
    
    # Risk indicators
    df_eng['high_debt_indicator'] = (df_eng['debt_to_income_ratio'] > 0.3).astype(int)
    df_eng['low_balance_indicator'] = (df_eng['account_balance'] < 5000).astype(int)
    df_eng['new_customer'] = (df_eng['customer_tenure'] < 1).astype(int)
    
    return df_eng

from sklearn.model_selection import train_test_split, cross_val_score, StratifiedKFold
from sklearn.ensemble import RandomForestClassifier, GradientBoostingClassifier
from sklearn.linear_model import LogisticRegression
from sklearn.svm import SVC
from sklearn.neighbors import KNeighborsClassifier
from sklearn.metrics import classification_report, confusion_matrix, roc_auc_score

class ModelComparison:
    def __init__(self):
        self.models = {}
        self.results = {}
        self.best_model = None
    
    def prepare_data(self, df, target_col='default_risk'):
        """Prepare data for modeling"""
        # Separate features and target
        X = df.drop(['customer_id', target_col], axis=1)
        y = df[target_col]
        
        # Preprocess features
        preprocessor = BankingDataPreprocessor()
        preprocessing_pipeline = preprocessor.create_preprocessing_pipeline(X)
        X_processed = preprocessor.fit_transform(X)
        
        # Split data
        X_train, X_test, y_train, y_test = train_test_split(
            X_processed, y, test_size=0.2, random_state=42, stratify=y
        )
        
        return X_train, X_test, y_train, y_test, preprocessor
    
    def initialize_models(self):
        """Initialize models for comparison"""
        self.models = {
            'Logistic Regression': LogisticRegression(random_state=42, max_iter=1000),
            'Random Forest': RandomForestClassifier(random_state=42, n_estimators=100),
            'Gradient Boosting': GradientBoostingClassifier(random_state=42, n_estimators=100),
            'SVM': SVC(random_state=42, probability=True),
            'K-Nearest Neighbors': KNeighborsClassifier(n_neighbors=5)
        }
    
    def evaluate_models(self, X_train, X_test, y_train, y_test):
        """Evaluate all models with cross-validation"""
        cv = StratifiedKFold(n_splits=5, shuffle=True, random_state=42)
        
        for name, model in self.models.items():
            print(f"\n=== Evaluating {name} ===")
            
            # Cross-validation scores
            cv_scores = cross_val_score(model, X_train, y_train, cv=cv, scoring='roc_auc')
            
            # Train model and predict
            model.fit(X_train, y_train)
            y_pred = model.predict(X_test)
            y_pred_proba = model.predict_proba(X_test)[:, 1]
            
            # Calculate metrics
            auc_score = roc_auc_score(y_test, y_pred_proba)
            
            self.results[name] = {
                'cv_auc_mean': cv_scores.mean(),
                'cv_auc_std': cv_scores.std(),
                'test_auc': auc_score,
                'model': model
            }
            
            print(f"CV AUC: {cv_scores.mean():.4f} (+/- {cv_scores.std() * 2:.4f})")
            print(f"Test AUC: {auc_score:.4f}")
            print("\nClassification Report:")
            print(classification_report(y_test, y_pred))
    
    def get_best_model(self):
        """Get the best performing model"""
        best_name = max(self.results.keys(), key=lambda k: self.results[k]['test_auc'])
        self.best_model = self.results[best_name]['model']
        return best_name, self.best_model
    
    def plot_model_comparison(self):
        """Plot model comparison results"""
        plt.figure(figsize=(12, 8))
        
        models = list(self.results.keys())
        cv_scores = [self.results[model]['cv_auc_mean'] for model in models]
        test_scores = [self.results[model]['test_auc'] for model in models]
        cv_stds = [self.results[model]['cv_auc_std'] for model in models]
        
        x = np.arange(len(models))
        width = 0.35
        
        plt.subplot(2, 1, 1)
        plt.bar(x - width/2, cv_scores, width, label='CV AUC', yerr=cv_stds, capsize=5)
        plt.bar(x + width/2, test_scores, width, label='Test AUC')
        plt.xlabel('Models')
        plt.ylabel('AUC Score')
        plt.title('Model Performance Comparison')
        plt.xticks(x, models, rotation=45)
        plt.legend()
        plt.grid(True, alpha=0.3)
        
        # Feature importance for best model
        best_name, best_model = self.get_best_model()
        if hasattr(best_model, 'feature_importances_'):
            plt.subplot(2, 1, 2)
            importances = best_model.feature_importances_
            indices = np.argsort(importances)[::-1][:10]  # Top 10 features
            
            plt.bar(range(10), importances[indices])
            plt.title(f'Top 10 Feature Importances - {best_name}')
            plt.xlabel('Feature Index')
            plt.ylabel('Importance')
        
        plt.tight_layout()
        plt.show()

# Run model comparison
